import React from 'react';
import ReactDOM from 'react-dom';
import 'antd/dist/antd.css';
import './index.css';
import { Radio } from 'antd';

class App extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      value: 0,
    };
  }


  onChange = e => {
    console.log('radio checked', e.target.value);
    this.setState({
      value: e.target.value,
    });
    const interval = e.target.value;
    if (Number(interval) !== 0) {
      const intervalNum = Number(interval) * 1000;
      console.log(intervalNum);
        this.ticker = setInterval(()=>{console.log("hello")}, intervalNum);
    }else{
      clearInterval(this.ticker);
    }
  };

  render() {
    return (
      <>
      开启定时刷新：
      <Radio.Group onChange={this.onChange} value={this.state.value}>
        <Radio value={1}>1s</Radio>
        <Radio value={10}>10s</Radio>
        <Radio value={15}>15s</Radio>
        <Radio value={0}>不开启</Radio>
      </Radio.Group>
      </>
    );
  }
}

ReactDOM.render(<App />, document.getElementById('container'));
